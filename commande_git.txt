//Configuration globale (une seule fois)
git config --global user.name "TonNomGitHub"
git config --global user.email "ton@email.com"

git config --global color.diff auto
git config --global color.status auto
git config --global color.branch auto

git config --global core.editor vim
git config --global merge.tool vimdiff
//Initialiser un dépôt local
cd chemin/vers/ton/projet
git init
//Lier un dépôt distant
git remote add origin <URL_DU_DEPOT>
git remote add origin git@github.com:username/repo.git(en ssh)
//Configurer SSH (connexion sécurisée)
Générer une clé SSH
ssh-keygen -t ed25519 -C "ton@email.com"
Copier la clé publique (Windows)
clip < ~/.ssh/id_ed25519.pub
 Ensuite :
Va sur GitHub → Settings → SSH keys
Colle la clé
//Préparer la branche principale
git branch -M main
git push -u origin main(git push --set-upstream origin master
//work_flow
git status              # voir les changements
git add .               # ajouter les fichiers
git commit -m "message" # valider les changements
git push                # envoyer vers GitHub
//Récupérer un projet existant
git clone <URL_DU_DEPOT>
cd repo
//git log --oneline       # historique simplifié
git diff                # voir les différences
git pull                # récupérer les modifications
# créer une nouvelle branche
git branch nom_branche

# créer une branche ET se déplacer dessus
git checkout -b nom_branche

# (version moderne recommandée)
git switch -c nom_branche

# se déplacer sur une branche existante
git checkout nom_branche

# (version moderne)
git switch nom_branche

# afficher les branches locales
git branch

# afficher toutes les branches (locales + distantes)
git branch -av

# supprimer une branche (sécurisé)
git branch -d nom_branche

# supprimer une branche en force
git branch -D nom_branche

# sauvegarder les modifications sans commit
git stash

# restaurer les modifications sauvegardées
git stash pop

# afficher la liste des stash
git stash list

# afficher l'historique simplifié
git log --oneline

# afficher l'historique avec graphe (très utile)
git log --oneline --graph --all

# voir les différences non commit
git diff
